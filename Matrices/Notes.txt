Reading:

int a[n][m];
for(int i=0; i<n; i++)
{
    for(int j=0; j<m; j++)
    {
        cin >> a[i][j];
    }
}

Printing:

for(int i=0; i<n; i++)
{
    for(int j=0; j<m; j++)
    {
        cout << a[i][j] << " ";
    }
    cout << endl;
}

Finding the minimum value:

// We assume the reading is already done

int min = a[0][0];
for(int i=0; i<n; i++)
{
    min = a[i][0];
    for(int j=0; j<m; j++)
    {
        if(a[i][j]<min)
        {
            min = a[i][j];
        }
    }
}
